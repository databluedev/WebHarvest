API

Firecrawl-Compatible

Same API contract. Your existing Firecrawl integrations work out of the box ‚Äî just change the base URL.

Request
import requests

response = requests.post(
    "http://localhost:8000/v1/crawl",
    headers={
        "Authorization": "Bearer wh_live_..."
    },
    json={
        "url": "https://docs.example.com",
        "maxDepth": 3,
        "maxPages": 100,
        "scrapeOptions": {
            "formats": ["markdown"],
            "onlyMainContent": True
        }
    }
)
# Returns crawl job ID for async polling
job = response.json()
print(job["id"])  # "9e6c8b0b-..."
